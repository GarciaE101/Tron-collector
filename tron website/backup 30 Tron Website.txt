        let sectionCount = 0;

        sectionCards.forEach(card => {
          const title = card.querySelector(".panel-title");
          const id = title.dataset.panel;
          const price = parseFloat(userPrices[id]) || 0;

          if (collected[id]) {
            sectionCount++;
            totalValue += price;
            card.classList.add("is-collected");
            title.classList.add("collected");
          } else {
            card.classList.remove("is-collected");
            title.classList.remove("collected");
          }
        });

        // Update Section Badge
        const sectionTotal = sectionCards.length;
        const sectionPercent = sectionTotal > 0 ? Math.round((sectionCount / sectionTotal) * 100) : 0;
        const badge = h2.querySelector(".percent-display");
        if (badge) {
          badge.textContent = `[ ${sectionPercent}% ]`;
          badge.style.color = sectionPercent === 100 ? "#39ff14" : "var(--accent)";
        }

        totalItems += sectionTotal;
        totalCollected += sectionCount;
      }
    });

    // Update Global Bar and Worth
    const globalPercent = totalItems > 0 ? Math.round((totalCollected / totalItems) * 100) : 0;
    document.getElementById("progress-bar-fill").style.width = `${globalPercent}%`;
    document.getElementById("total-percent").textContent = `${globalPercent}%`;
    document.getElementById("total-worth").textContent = `$${totalValue.toFixed(2)}`;
  };

  // Attach listeners to every card
  document.querySelectorAll(".card").forEach(card => {
    const title = card.querySelector(".panel-title");
    const id = title.dataset.panel;
    const select = card.querySelector(".source-select");
    const priceInput = card.querySelector(".price-input");

    // Restore saved values
    if (sources[id]) select.value = sources[id];
    if (userPrices[id]) priceInput.value = userPrices[id];

    // Click Image to Collect
    card.querySelector("img").addEventListener("click", () => {
      collected[id] = !collected[id];
      if (!collected[id]) delete collected[id];
      localStorage.setItem("tronCollected", JSON.stringify(collected));
      updateProgress();
    });

    // Change Dropdown
    select.addEventListener("change", (e) => {
      sources[id] = e.target.value;
      localStorage.setItem("tronSources", JSON.stringify(sources));
    });

    // Enter Price
    priceInput.addEventListener("input", (e) => {
      userPrices[id] = e.target.value;
      localStorage.setItem("tronPrices", JSON.stringify(userPrices));
      updateProgress();
    });
  });

  updateProgress();

  // Reset Everything
  document.getElementById("resetCollected").addEventListener("click", () => {
    if (confirm("De-rez all data? (Collection, Sources, and Prices)")) {
      localStorage.clear();
      location.reload();
    }
  });
});
