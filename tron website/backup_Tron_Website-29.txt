##js
document.addEventListener("DOMContentLoaded", () => {
  let collected = JSON.parse(localStorage.getItem("tronCollected")) || {};

  const titles = document.querySelectorAll(".panel-title");

  // Restore collected state
  titles.forEach(title => {
    const id = title.dataset.panel;
    if (collected[id]) {
      title.classList.add("collected");
    }
  });

  // Toggle collected on image click
  document.querySelectorAll(".card img").forEach(img => {
    img.addEventListener("click", () => {
      const title = img.closest(".card").querySelector(".panel-title");
      const id = title.dataset.panel;

      collected[id] = !collected[id];
      title.classList.toggle("collected");

      localStorage.setItem("tronCollected", JSON.stringify(collected));
    });
  });

  // 🔥 RESET ALL BUTTON
  const resetBtn = document.getElementById("resetCollected");

  resetBtn.addEventListener("click", () => {
    if (!confirm("Reset entire TRON collection?")) return;

    collected = {};
    localStorage.removeItem("tronCollected");

    titles.forEach(title => {
      title.classList.remove("collected");
    });
  });
});
